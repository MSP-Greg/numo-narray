static VALUE step_each( VALUE self ) {

VALUE  a;
double beg, step;
size_t i, size;

a = nary_step_parameters( self, Qnil );
beg  = NUM2DBL(RARRAY_PTR(a)[0]);
step = NUM2DBL(RARRAY_PTR(a)[1]);
size = NUM2SIZET(RARRAY_PTR(a)[2]);

for (i=0; i<size; i++) {
    rb_yield(rb_float_new(beg+i*step));
}
return self;

}
