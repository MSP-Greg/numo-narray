static VALUE
numo_dcomplex_s_cast(VALUE type, VALUE obj)
{
    VALUE v;
    narray_t *na;
    dtype x;

    if (CLASS_OF(obj)==cT) {
        return obj;
    }
    if (RTEST(rb_obj_is_kind_of(obj,rb_cNumeric))) {
        x = m_num_to_data(obj);
        return numo_dcomplex_new_dim0(x);
    }
    if (RTEST(rb_obj_is_kind_of(obj,rb_cArray))) {
        return numo_dcomplex_cast_array(obj);
    }
    if (IsNArray(obj)) {
        GetNArray(obj,na);
        v = rb_narray_new(cT, NA_NDIM(na), NA_SHAPE(na));
        if (NA_SIZE(na) > 0) {
            numo_dcomplex_allocate(v);
            numo_dcomplex_store(v,obj);
        }
        return v;
    }
    rb_raise(nary_eCastError,"cannot cast to %s",rb_class2name(type));
    return Qnil;
}

static VALUE
numo_dcomplex_aset(int argc, VALUE *argv, VALUE self)
{
    ssize_t pos;
    char *ptr;
    VALUE a;

    argc--;
    if (argc==0) {
        numo_dcomplex_store(self, argv[argc]);
    } else {
        pos = na_get_scalar_position(self, argc, argv, sizeof(dtype));
        if (pos == -1) {
            a = na_aref_main(argc, argv, self, 0);
            numo_dcomplex_store(a, argv[argc]);
        } else {
            ptr = na_get_pointer_for_read(self) + pos;
            *(dtype*)ptr = m_num_to_data(argv[argc]);
        }

    }
    return argv[argc];
}
